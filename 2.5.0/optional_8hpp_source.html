212 [[nodiscard]] detail::argument_descriptor desc(const bool verbose) const noexcept override {
213 detail::argument_descriptor desc(this->_name.str(), this->_help_msg);
214
215 if (not verbose)
216 return desc;
217
218 desc.params.reserve(6);
219 if (this->_required)
220 desc.add_param("required", "true");
221 if (this->bypass_required_enabled())
222 desc.add_param("bypass required", "true");
223 if (this->_nargs_range.is_bound())
224 desc.add_param("nargs", this->_nargs_range);
225 if constexpr (detail::c_writable<value_type>) {
226 if (not this->_choices.empty())
227 desc.add_range_param("choices", this->_choices);
228 if (this->_default_value.has_value())
229 desc.add_param("default value", std::any_cast<value_type>(this->_default_value));
230 if (this->_implicit_value.has_value())
231 desc.add_param("implicit value", std::any_cast<value_type>(this->_implicit_value));
232 }

261 bool set_value(const std::string& str_value) override {
262 if (not this->_accepts_further_values())
263 throw parsing_failure::invalid_nvalues(this->_name, std::weak_ordering::greater);
264
265 value_type value;
266 if constexpr (detail::c_trivially_readable<value_type>) {
267 value = value_type(str_value);
268 }
269 else {
270 if (not (std::istringstream(str_value) >> value))
271 throw parsing_failure::invalid_value(this->_name, str_value);
272 }
273
274 if (not detail::is_valid_choice(value, this->_choices))
275 throw parsing_failure::invalid_choice(this->_name, str_value);
276
277 const auto apply_visitor = action::detail::apply_visitor<value_type>{value};
278 for (const auto& action : this->_value_actions)
279 std::visit(apply_visitor, action);
280
281 this->_values.emplace_back(std::move(value));
282 return this->_accepts_further_values();
283 }
284
286 [[nodiscard]] bool has_value() const noexcept override {

288 }

314 [[nodiscard]] bool _has_predefined_value() const noexcept {

318
323 [[nodiscard]] const std::any& _predefined_value() const {
324 if (this->is_used()) {
325 if (not this->_implicit_value.has_value())
326 throw(std::logic_error(
327 std::format("No implicit value specified for argument `{}`.", this->_name.str())
328 ));
329
330 return this->_implicit_value;
331 }
332
333 if (not this->_default_value.has_value())
334 throw(std::logic_error(
335 std::format("No default value specified for argument `{}`.", this->_name.str())
336 ));
337
338 return this->_default_value;
339 }
340
341 [[nodiscard]] bool _accepts_further_values() const noexcept {
342 return not std::is_gt(this->_nargs_range.ordering(this->_values.size() + 1ull));
343 }
