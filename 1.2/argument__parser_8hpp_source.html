114struct argument_name {
115 argument_name() = delete;
116 argument_name& operator=(const argument_name&) = delete;
117 argument_name& operator=(argument_name&&) = delete;
118
120 argument_name(const argument_name&) = default;
121
123 argument_name(argument_name&&) = default;
124
129 argument_name(std::string_view primary) : primary(primary) {}
130
136 argument_name(std::string_view primary, std::string_view secondary)
137 : primary(primary), secondary(secondary) {}
138
140 ~argument_name() = default;
141
147 bool operator==(const argument_name& other) const noexcept {
148 if (not (this->secondary and other.secondary) and (this->secondary or other.secondary))
149 return false;
150
151 if (this->primary != other.primary)
152 return false;
153
154 return this->secondary ? this->secondary.value() == other.secondary.value() : true;
155 }
156
162 [[nodiscard]] bool match(std::string_view arg_name) const noexcept {
163 return arg_name == this->primary
164 or (this->secondary and arg_name == this->secondary.value());
165 }
166
172 [[nodiscard]] bool match(const argument_name& arg_name) const noexcept {
173 if (this->match(arg_name.primary))
174 return true;
175
176 if (arg_name.secondary)
177 return this->match(arg_name.secondary.value());
178
179 return false;
180 }
181
183 [[nodiscard]] std::string str() const noexcept {
184 return this->secondary
185 ? ("[" + this->primary + "," + this->secondary.value() + "]")
186 : ("[" + this->primary + "]");
187 }
188
195 friend std::ostream& operator<<(std::ostream& os, const argument_name& arg_name) noexcept {
196 os << arg_name.str();
197 return os;
198 }
199
200 const std::string primary;
201 const std::optional<std::string> secondary;
202};

454
460 range(const count_type nlow, const count_type nhigh)
461 : _nlow(nlow), _nhigh(nhigh), _default(nlow == _ndefault and nhigh == _ndefault) {}
462
464 range(const range&) = default;
465
467 range(range&&) = default;
468
470 range& operator=(const range&) = default;
471
473 range& operator=(range&&) = default;
474
476 ~range() = default;
477
479 [[nodiscard]] bool is_default() const noexcept {
480 return this->_default;
481 }
482
488 [[nodiscard]] std::weak_ordering contains(const range::count_type n) const noexcept {
489 if (not (this->_nlow.has_value() or this->_nhigh.has_value()))
490 return std::weak_ordering::equivalent;
491
492 if (this->_nlow.has_value() and this->_nhigh.has_value()) {
493 if (n < this->_nlow.value())
494 return std::weak_ordering::less;
495
496 if (n > this->_nhigh.value())
497 return std::weak_ordering::greater;
498
499 return std::weak_ordering::equivalent;
500 }
501
502 if (this->_nlow.has_value())
503 return (n < this->_nlow.value())
504 ? std::weak_ordering::less
505 : std::weak_ordering::equivalent;
506
507 return (n > this->_nhigh.value())
508 ? std::weak_ordering::greater
509 : std::weak_ordering::equivalent;
510 }
511
512 friend range at_least(const count_type) noexcept;
513 friend range more_than(const count_type) noexcept;
514 friend range less_than(const count_type) noexcept;
515 friend range up_to(const count_type) noexcept;
516 friend range any() noexcept;
517
518private:
524 range(const std::optional<count_type> nlow, const std::optional<count_type> nhigh)
525 : _nlow(nlow), _nhigh(nhigh) {}
526
527 std::optional<count_type> _nlow;
528 std::optional<count_type> _nhigh;
529 bool _default = true;
530
531 static constexpr count_type _ndefault = 1;
532};

606template <typename AS>
607concept c_action_specifier = ap::utility::is_valid_type_v<AS, ap::valued_action, ap::void_action>;
608
610template <c_action_specifier AS, ap::utility::c_argument_value_type T>
611using callable_type = typename AS::template type<T>;
612
614template <ap::utility::c_argument_value_type T>
615using action_variant_type =
616 std::variant<callable_type<ap::valued_action, T>, callable_type<ap::void_action, T>>;
617
624template <ap::utility::c_argument_value_type T>
625[[nodiscard]] inline bool is_void_action(const action_variant_type<T>& action) noexcept {
626 return std::holds_alternative<callable_type<ap::void_action, T>>(action);
627}
628
629} // namespace detail
