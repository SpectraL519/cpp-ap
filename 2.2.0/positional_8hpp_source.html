186 positional& set_value(const std::string& str_value) override {
187 if (this->_value.has_value())
188 throw error::value_already_set(this->_name);
189
190 this->_ss.clear();
191 this->_ss.str(str_value);
192
193 value_type value;
194 if (not (this->_ss >> value))
195 throw error::invalid_value(this->_name, str_value);
196
197 if (not this->_is_valid_choice(value))
198 throw error::invalid_choice(this->_name, str_value);
199
200 const auto apply_visitor = action::detail::apply_visitor<value_type>{value};
201 for (const auto& action : this->_value_actions)
202 std::visit(apply_visitor, action);
203
204 this->_value = value;
205 return *this;
206 }
