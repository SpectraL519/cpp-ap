231 optional& set_value(const std::string& str_value) override {
232 this->_ss.clear();
233 this->_ss.str(str_value);
234
235 value_type value;
236 if (not (this->_ss >> value))
237 throw error::invalid_value(this->_name, this->_ss.str());
238
239 if (not this->_is_valid_choice(value))
240 throw error::invalid_choice(this->_name, str_value);
241
242 this->_apply_action(value);
243
244 if (not (this->_nargs_range or this->_values.empty()))
245 throw error::value_already_set(this->_name);
246
247 this->_values.emplace_back(std::move(value));
248 return *this;
249 }

322 void _apply_action(value_type& value) const noexcept {
323 if (action::detail::is_modify_action(this->_action)) {

326 std::get<callable_type>(this->_action)(value);
327 }
328 else {
329 using callable_type =
330 action::detail::callable_type<ap::action_type::transform, value_type>;
331 value = std::get<callable_type>(this->_action)(value);
332 }
