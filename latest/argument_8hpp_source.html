458 [[nodiscard]] detail::help_builder help_builder(const bool verbose) const noexcept override {
459 detail::help_builder bld(this->_name.str(), this->_help_msg);
460
461 if (not verbose)
462 return bld;
463
464 bld.params.reserve(6ull);
465 if (this->_required != _default_required)
466 bld.add_param("required", std::format("{}", this->_required));
467 if (this->_suppress_arg_checks)
468 bld.add_param("suppress arg checks", "true");
469 if (this->_suppress_group_checks)
470 bld.add_param("suppress group checks", "true");
471 if (this->_nargs_range != _default_nargs_range)
472 bld.add_param("nargs", this->_nargs_range);
473 if constexpr (util::c_writable<value_type>) {
474 if (not this->_choices.empty())
475 bld.add_range_param("choices", this->_choices);
476 if (not this->_default_values.empty())
477 bld.add_range_param(
478 "default value(s)", util::any_range_cast_view<value_type>(this->_default_values)
479 );
480 if constexpr (type == argument_type::optional) {
481 if (not this->_implicit_values.empty())
482 bld.add_range_param(
483 "implicit value(s)",
484 util::any_range_cast_view<value_type>(this->_implicit_values)
485 );
486 }
487 }
488
489 return bld;
490 }

684
695 bool _set_value_impl(const std::string& str_value)
696 requires(not util::c_is_none<value_type>)
697 {
698 if (not this->_accepts_further_values())
699 throw parsing_failure::invalid_nvalues(this->_name, std::weak_ordering::greater);
700
701 value_type value;
702 if constexpr (util::c_trivially_readable<value_type>) {
703 value = value_type(str_value);
704 }
705 else {
706 if (not (std::istringstream(str_value) >> value))
707 throw parsing_failure(std::format(
708 "Cannot parse value `{}` for argument [{}].", str_value, this->_name.str()
709 ));
710 }
711
712 if (not this->_is_valid_choice(value))
713 throw parsing_failure(std::format(
714 "Value `{}` is not a valid choice for argument [{}].", str_value, this->_name.str()
715 ));
716
717 const auto apply_visitor = action::util::apply_visitor<value_type>{value};
718 for (const auto& action : this->_value_actions)
719 std::visit(apply_visitor, action);
720
721 this->_values.emplace_back(std::move(value));
722 return this->_accepts_further_values();
723 }
