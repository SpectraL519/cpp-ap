161 [[nodiscard]] detail::argument_descriptor desc(const bool verbose) const noexcept override {
162 detail::argument_descriptor desc(this->_name.str(), this->_help_msg);
163
164 if (not verbose)
165 return desc;
166
167 if (not this->_required)
168 desc.add_param("required", "false");
169 if (this->bypass_required_enabled())

171 if constexpr (detail::c_writable<value_type>) {

174 if (this->_default_value.has_value())

205 bool set_value(const std::string& str_value) override {
206 if (this->_value.has_value())
207 throw parsing_failure::value_already_set(this->_name);
208
209 value_type value;
210 if constexpr (detail::c_trivially_readable<value_type>) {
211 value = value_type(str_value);
212 }
213 else {
214 if (not (std::istringstream(str_value) >> value))
215 throw parsing_failure::invalid_value(this->_name, str_value);
216 }
217
218 if (not detail::is_valid_choice(value, this->_choices))
219 throw parsing_failure::invalid_choice(this->_name, str_value);
220
221 const auto apply_visitor = action::detail::apply_visitor<value_type>{value};
222 for (const auto& action : this->_value_actions)
223 std::visit(apply_visitor, action);
224
225 this->_value = value;
226 return false;
227 }
228
230 [[nodiscard]] bool has_value() const noexcept override {

232 }
