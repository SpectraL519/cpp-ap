273 template <detail::c_range_of<std::string, detail::type_validator::convertible> AR>
274 void parse_args(const AR& argv) {
275 this->_validate_argument_configuration();
276 this->_parse_args_impl(this->_tokenize(argv));
277
278 if (this->_are_required_args_bypassed())
279 return;
280
281 this->_verify_required_args();
282 this->_verify_nvalues();

357 template <detail::c_argument_value_type T = std::string>
358 [[nodiscard]] T value(std::string_view arg_name) const {
359 const auto arg_opt = this->_get_argument(arg_name);
360 if (not arg_opt)
361 throw lookup_failure::argument_not_found(arg_name);
362
363 const auto& arg_value = arg_opt->get().value();
364 try {
365 return std::any_cast<T>(arg_value);
366 }
367 catch (const std::bad_any_cast&) {
368 throw type_error::invalid_value_type(arg_opt->get().name(), typeid(T));
369 }

380 template <detail::c_argument_value_type T = std::string, std::convertible_to<T> U>
381 [[nodiscard]] T value_or(std::string_view arg_name, U&& fallback_value) const {
382 const auto arg_opt = this->_get_argument(arg_name);
383 if (not arg_opt)
384 throw lookup_failure::argument_not_found(arg_name);
385
386 try {
387 const auto& arg_value = arg_opt->get().value();
388 return std::any_cast<T>(arg_value);
389 }
390 catch (const std::logic_error&) {
391 // positional: no value parsed
392 // optional: no value parsed + no predefined value
393 return T{std::forward<U>(fallback_value)};
394 }
395 catch (const std::bad_any_cast&) {
396 throw type_error::invalid_value_type(arg_opt->get().name(), typeid(T));
397 }
398 }

406 template <detail::c_argument_value_type T = std::string>
407 [[nodiscard]] std::vector<T> values(std::string_view arg_name) const {
408 const auto arg_opt = this->_get_argument(arg_name);
409 if (not arg_opt)
410 throw lookup_failure::argument_not_found(arg_name);
411
412 const auto& arg = arg_opt->get();
413
414 try {
415 if (not arg.has_parsed_values() and arg.has_value())
416 return std::vector<T>{std::any_cast<T>(arg.value())};
417
418 std::vector<T> values;
419 // TODO: use std::ranges::to after transition to C++23
420 std::ranges::copy(
421 std::views::transform(
422 arg.values(), [](const std::any& value) { return std::any_cast<T>(value); }
423 ),
424 std::back_inserter(values)
425 );
426 return values;
427 }
428 catch (const std::bad_any_cast&) {
429 throw type_error::invalid_value_type(arg.name(), typeid(T));
430 }

492 void _verify_arg_name_pattern(const std::string_view arg_name) const {
493 if (arg_name.empty())
494 throw invalid_configuration::invalid_argument_name(
495 arg_name, "An argument name cannot be empty."
496 );
497
498 if (detail::contains_whitespaces(arg_name))
499 throw invalid_configuration::invalid_argument_name(
500 arg_name, "An argument name cannot contain whitespaces."
501 );
502
503 if (arg_name.front() == this->_flag_prefix_char)
504 throw invalid_configuration::invalid_argument_name(
505 arg_name,
506 std::format(
507 "An argument name cannot begin with a flag prefix character ({}).",
508 this->_flag_prefix_char
509 )
510 );
511
512 if (std::isdigit(arg_name.front()))
513 throw invalid_configuration::invalid_argument_name(
514 arg_name, "An argument name cannot begin with a digit."
515 );
516 }

552 [[nodiscard]] bool _is_arg_name_used(
553 const detail::argument_name& arg_name,
554 const detail::argument_name::match_type m_type = detail::argument_name::m_any
555 ) const noexcept {
556 const auto predicate = this->_name_match_predicate(arg_name, m_type);
557
558 if (std::ranges::find_if(this->_positional_args, predicate) != this->_positional_args.end())
559 return true;
560
561 if (std::ranges::find_if(this->_optional_args, predicate) != this->_optional_args.end())
562 return true;
563
564 return false;

573 void _validate_argument_configuration() const {
574 // step 1
575 const_arg_opt_t non_required_arg = std::nullopt;
576 for (const auto& arg : this->_positional_args) {
577 if (not arg->is_required()) {
578 non_required_arg = std::ref(*arg);
579 continue;
580 }
581
582 if (non_required_arg and arg->is_required())
583 throw invalid_configuration::positional::required_after_non_required(
584 arg->name(), non_required_arg->get().name()
585 );
586 }

595 template <detail::c_sized_range_of<std::string_view, detail::type_validator::convertible> AR>
596 [[nodiscard]] arg_token_list_t _tokenize(const AR& arg_range) const {
597 const auto n_args = std::ranges::size(arg_range);
598 if (n_args == 0ull)
599 return arg_token_list_t{};
600
601 arg_token_list_t toks;
602 toks.reserve(n_args);
603 std::ranges::for_each(
604 arg_range, std::bind_front(&argument_parser::_tokenize_arg, this, std::ref(toks))
605 );
606 return toks;
607 }
608
614 void _tokenize_arg(arg_token_list_t& toks, const std::string_view arg_value) const {
615 auto tok = this->_build_token(arg_value);
616 if (tok.is_flag_token() and not this->_is_valid_flag(tok)) {
617#ifdef AP_UNKNOWN_FLAGS_AS_VALUES
618 toks.emplace_back(detail::argument_token::t_value, std::string(arg_value));
619 return;
620#else
621 throw parsing_failure::unknown_argument(arg_value);
622#endif
623 }
624
625 toks.emplace_back(std::move(tok));
626 }

633 [[nodiscard]] detail::argument_token _build_token(const std::string_view arg_value
634 ) const noexcept {
635 if (detail::contains_whitespaces(arg_value))
636 return {.type = detail::argument_token::t_value, .value = std::string(arg_value)};
637
638 if (arg_value.starts_with(this->_flag_prefix))
639 return {
640 .type = detail::argument_token::t_flag_primary,
641 .value = std::string(arg_value.substr(this->_primary_flag_prefix_length))
642 };
643
644 if (arg_value.starts_with(this->_flag_prefix_char))
645 return {
646 .type = detail::argument_token::t_flag_secondary,
647 .value = std::string(arg_value.substr(this->_secondary_flag_prefix_length))
648 };
649
650 return {.type = detail::argument_token::t_value, .value = std::string(arg_value)};
651 }

658 [[nodiscard]] bool _is_valid_flag(const detail::argument_token& tok) const noexcept {
659 if (tok.type == detail::argument_token::t_flag_primary)
660 return this->_is_arg_name_used(
661 detail::argument_name{tok.value}, detail::argument_name::m_primary
662 );
663
664 if (tok.type == detail::argument_token::t_flag_secondary)
665 return this->_is_arg_name_used(
666 detail::argument_name{tok.value}, detail::argument_name::m_secondary
667 );
668
669 return false;
670 }

684 [[nodiscard]] std::string _unstripped_token_value(const detail::argument_token& tok
685 ) const noexcept {
686 switch (tok.type) {
687 case detail::argument_token::t_flag_primary:
688 return std::format("{}{}", this->_flag_prefix, tok.value);
689 case detail::argument_token::t_flag_secondary:
690 return std::format("{}{}", this->_flag_prefix_char, tok.value);
691 default:
692 return tok.value;
693 }

702 void _parse_args_impl(const arg_token_list_t& arg_tokens) {
703 arg_token_list_iterator_t token_it = arg_tokens.begin();
704
705 this->_parse_positional_args(token_it, arg_tokens.end());
706
707 std::vector<std::string_view> unknown_args;
708 this->_parse_optional_args(token_it, arg_tokens.end(), unknown_args);
709
710 if (not unknown_args.empty())
711 throw parsing_failure::argument_deduction_failure(unknown_args);
712 }
713
719 void _parse_positional_args(
720 arg_token_list_iterator_t& token_it, const arg_token_list_iterator_t& tokens_end
721 ) noexcept {
722 for (const auto& pos_arg : this->_positional_args) {
723 if (token_it == tokens_end)
724 return;
725
726 if (token_it->type != detail::argument_token::t_value)
727 return;
728
729 pos_arg->set_value(token_it->value);
730 ++token_it;
731 }
732 }
733
742 void _parse_optional_args(
743 arg_token_list_iterator_t& token_it,
744 const arg_token_list_iterator_t& tokens_end,
745 std::vector<std::string_view>& unknown_args
746 ) {
747 std::optional<std::reference_wrapper<arg_ptr_t>> curr_opt_arg;
748
749 while (token_it != tokens_end) {
750 switch (token_it->type) {
751 case detail::argument_token::t_flag_primary:
752 [[fallthrough]];
753 case detail::argument_token::t_flag_secondary: {
754 // TODO: remove
755 if (not unknown_args.empty())
756 throw parsing_failure::argument_deduction_failure(unknown_args);
757
758 const auto opt_arg_it = this->_find_opt_arg(*token_it);
759 if (opt_arg_it == this->_optional_args.end())
760 throw parsing_failure::unknown_argument(this->_unstripped_token_value(*token_it)
761 );
762
763 curr_opt_arg = std::ref(*opt_arg_it);
764 if (not curr_opt_arg->get()->mark_used())
765 curr_opt_arg.reset();
766
767 break;
768 }
769 case detail::argument_token::t_value: {
770 if (not curr_opt_arg) {
771 unknown_args.emplace_back(token_it->value);
772 break;
773 }
774
775 if (not curr_opt_arg->get()->set_value(token_it->value))
776 curr_opt_arg.reset();
777
778 break;
779 }
780 }
781
782 ++token_it;
783 }
784 }

790 [[nodiscard]] bool _are_required_args_bypassed() const noexcept {
791 // TODO: use std::views::join after the transition to C++23
792 return std::ranges::any_of(
793 this->_positional_args,
794 [](const arg_ptr_t& arg) {
795 return arg->is_used() and arg->bypass_required_enabled();
796 }
797 )
798 or std::ranges::any_of(this->_optional_args, [](const arg_ptr_t& arg) {
799 return arg->is_used() and arg->bypass_required_enabled();
800 });
801 }
802
807 void _verify_required_args() const {
808 // TODO: use std::views::join after the transition to C++23
809 for (const auto& arg : this->_positional_args)
810 if (arg->is_required() and not arg->has_value())
811 throw parsing_failure::required_argument_not_parsed(arg->name());
812
813 for (const auto& arg : this->_optional_args)
814 if (arg->is_required() and not arg->has_value())
815 throw parsing_failure::required_argument_not_parsed(arg->name());
816 }
817
822 void _verify_nvalues() const {
823 // TODO: use std::views::join after the transition to C++23
824 for (const auto& arg : this->_positional_args)
825 if (const auto nv_ord = arg->nvalues_ordering(); not std::is_eq(nv_ord))
826 throw parsing_failure::invalid_nvalues(arg->name(), nv_ord);
827
828 for (const auto& arg : this->_optional_args)
829 if (const auto nv_ord = arg->nvalues_ordering(); not std::is_eq(nv_ord))
830 throw parsing_failure::invalid_nvalues(arg->name(), nv_ord);
831 }
832
838 arg_opt_t _get_argument(std::string_view arg_name) const noexcept {
839 const auto predicate = this->_name_match_predicate(arg_name);
840
841 if (auto pos_arg_it = std::ranges::find_if(this->_positional_args, predicate);
842 pos_arg_it != this->_positional_args.end()) {
843 return std::ref(**pos_arg_it);
844 }
845
846 if (auto opt_arg_it = std::ranges::find_if(this->_optional_args, predicate);
847 opt_arg_it != this->_optional_args.end()) {
848 return std::ref(**opt_arg_it);
849 }
850
851 return std::nullopt;
852 }
853
860 [[nodiscard]] arg_ptr_list_iter_t _find_opt_arg(const detail::argument_token& flag_tok
861 ) noexcept {
862 switch (flag_tok.type) {
863 case detail::argument_token::t_flag_primary:
864 return std::ranges::find(this->_optional_args, flag_tok.value, [](const auto& arg_ptr) {
865 return arg_ptr->name().primary;
866 });
867 case detail::argument_token::t_flag_secondary:
868 return std::ranges::find(this->_optional_args, flag_tok.value, [](const auto& arg_ptr) {
869 return arg_ptr->name().secondary;
870 });
871 default:
872 return this->_optional_args.end();
873 }
874 }
875
881 void _print(std::ostream& os, const arg_ptr_list_t& args, const bool verbose) const noexcept {
882 if (verbose) {
883 for (const auto& arg : args)
884 os << '\n' << arg->desc(verbose).get(this->_indent_width) << '\n';
885 }
886 else {
887 std::vector<detail::argument_descriptor> descriptors;
888 descriptors.reserve(args.size());
889
890 for (const auto& arg : args)
891 descriptors.emplace_back(arg->desc(verbose));
892
893 std::size_t max_arg_name_length = 0ull;
894 for (const auto& desc : descriptors)
895 max_arg_name_length = std::max(max_arg_name_length, desc.name.length());
896
897 for (const auto& desc : descriptors)
898 os << '\n' << desc.get_basic(this->_indent_width, max_arg_name_length);
899
900 os << '\n';
901 }
902 }

925inline void add_default_argument(
926 const argument::default_positional arg_discriminator, argument_parser& arg_parser
927) noexcept {
928 switch (arg_discriminator) {
929 case argument::default_positional::input:
930 arg_parser.add_positional_argument("input")
931 .action<action_type::observe>(action::check_file_exists())
932 .help("Input file path");
933 break;
934
935 case argument::default_positional::output:
936 arg_parser.add_positional_argument("output").help("Output file path");
