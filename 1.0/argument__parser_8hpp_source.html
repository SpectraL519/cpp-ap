116class range {
117public:
118 using count_type = std::size_t;
119
121 range() : _nlow(_ndefault), _nhigh(_ndefault) {}
122
127 range(const count_type n)
128 : _nlow(n), _nhigh(n), _default(n == _ndefault) {}
129
135 range(const count_type nlow, const count_type nhigh)
136 : _nlow(nlow), _nhigh(nhigh), _default(nlow == _ndefault and nhigh == _ndefault) {}
137
142 range& operator=(const range&) = default;
143
145 ~range() = default;
146
148 [[nodiscard]] inline bool is_default() const {
149 return this->_default;
150 }
151
157 [[nodiscard]] std::weak_ordering contains(const range::count_type n) const {
158 if (not (this->_nlow.has_value() or this->_nhigh.has_value()))
159 return std::weak_ordering::equivalent;
160
161 if (this->_nlow.has_value() and this->_nhigh.has_value()) {
162 if (n < this->_nlow.value())
163 return std::weak_ordering::less;
164
165 if (n > this->_nhigh.value())
166 return std::weak_ordering::greater;
167
168 return std::weak_ordering::equivalent;
169 }
170
171 if (this->_nlow.has_value())
172 return (n < this->_nlow.value()) ? std::weak_ordering::less
173 : std::weak_ordering::equivalent;
174
175 return (n > this->_nhigh.value()) ? std::weak_ordering::greater
176 : std::weak_ordering::equivalent;
177 }
178
179 friend range at_least(const count_type);
180 friend range more_than(const count_type);
181 friend range less_than(const count_type);
182 friend range up_to(const count_type);
183 friend range any();
184
185private:
191 range(const std::optional<count_type> nlow, const std::optional<count_type> nhigh)
192 : _nlow(nlow), _nhigh(nhigh) {}
193
194 std::optional<count_type> _nlow;
195 std::optional<count_type> _nhigh;
196 bool _default = true;
197
198 static constexpr count_type _ndefault = 1;
199};

1744 throw error::free_value_error(cmd_it->value);

1746 curr_opt_arg->get()->set_value(cmd_it->value);
