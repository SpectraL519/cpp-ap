341 template <detail::c_argument_value_type T = std::string>
342 T value(std::string_view arg_name) const {
343 const auto arg_opt = this->_get_argument(arg_name);
344 if (not arg_opt)
345 throw error::argument_not_found(arg_name);
346
347 const auto& arg_value = arg_opt->get().value();
348 try {
349 return std::any_cast<T>(arg_value);
350 }
351 catch (const std::bad_any_cast& err) {
352 throw error::invalid_value_type(arg_opt->get().name(), typeid(T));
353 }
354 }

365 template <detail::c_argument_value_type T = std::string, std::convertible_to<T> U>
366 T value_or(std::string_view arg_name, U&& default_value) const {
367 const auto arg_opt = this->_get_argument(arg_name);
368 if (not arg_opt)
369 throw error::argument_not_found(arg_name);
370
371 try {
372 const auto& arg_value = arg_opt->get().value();
373 return std::any_cast<T>(arg_value);
374 }
375 catch (const std::logic_error&) {
376 // positional: no value parsed
377 // optional: no value parsed + no predefined value
378 return T{std::forward<U>(default_value)};
379 }
380 catch (const std::bad_any_cast& err) {
381 throw error::invalid_value_type(arg_opt->get().name(), typeid(T));
382 }
383 }

392 template <detail::c_argument_value_type T = std::string>
393 std::vector<T> values(std::string_view arg_name) const {
394 const auto arg_opt = this->_get_argument(arg_name);
395 if (not arg_opt)
396 throw error::argument_not_found(arg_name);
397
398 const auto& arg = arg_opt->get();
399
400 try {
401 if (not arg.has_parsed_values() and arg.has_value())
402 return std::vector<T>{std::any_cast<T>(arg.value())};
403
404 std::vector<T> values;
405 // TODO: use std::ranges::to after transition to C++23
406 std::ranges::copy(
407 std::views::transform(
408 arg.values(), [](const std::any& value) { return std::any_cast<T>(value); }
409 ),
410 std::back_inserter(values)
411 );
412 return values;
413 }
414 catch (const std::bad_any_cast& err) {
415 throw error::invalid_value_type(arg.name(), typeid(T));
416 }

458 void _verify_arg_name_pattern(const std::string_view arg_name) const {
459 if (arg_name.empty())
460 throw error::invalid_argument_name_pattern(
461 arg_name, "An argument name cannot be empty."
462 );
463
464 if (arg_name.front() == this->_flag_prefix_char)
465 throw error::invalid_argument_name_pattern(
466 arg_name,
467 std::format(
468 "An argument name cannot begin with a flag prefix character ({}).",
469 this->_flag_prefix_char
470 )
471 );
472
473 if (std::isdigit(arg_name.front()))
474 throw error::invalid_argument_name_pattern(
475 arg_name, "An argument name cannot begin with a digit."
476 );
477 }

484 [[nodiscard]] auto _name_match_predicate(const std::string_view arg_name) const noexcept {
485 return [arg_name](const arg_ptr_t& arg) { return arg->name().match(arg_name); };
486 }

493 [[nodiscard]] auto _name_match_predicate(const detail::argument_name& arg_name) const noexcept {
494 return [&arg_name](const arg_ptr_t& arg) { return arg->name().match(arg_name); };
495 }

502 [[nodiscard]] bool _is_arg_name_used(const detail::argument_name& arg_name) const noexcept {
503 const auto predicate = this->_name_match_predicate(arg_name);
504
505 if (std::ranges::find_if(this->_positional_args, predicate) != this->_positional_args.end())
506 return true;
507
508 if (std::ranges::find_if(this->_optional_args, predicate) != this->_optional_args.end())
509 return true;
510
511 return false;
512 }

520 template <detail::c_sized_range_of<std::string, detail::type_validator::convertible> AR>
521 [[nodiscard]] arg_token_list_t _tokenize(const AR& arg_range) const noexcept {
522 const auto n_args = std::ranges::size(arg_range);
523 if (n_args == 0ull)
524 return arg_token_list_t{};
525
526 arg_token_list_t toks;
527 toks.reserve(n_args);
528
529 for (const auto& arg : arg_range) {
530 std::string value = static_cast<std::string>(arg);
531 if (this->_is_flag(value)) {
532 this->_strip_flag_prefix(value);
533 toks.emplace_back(detail::argument_token::t_flag, std::move(value));
534 }
535 else {
536 toks.emplace_back(detail::argument_token::t_value, std::move(value));
537 }
538 }
539
540 return toks;
541 }
542
548 [[nodiscard]] bool _is_flag(const std::string& arg) const noexcept {
549 if (arg.starts_with(this->_flag_prefix))
550 return this->_is_arg_name_used({arg.substr(this->_flag_prefix_length)});
551
552 if (arg.starts_with(this->_flag_prefix_char))
553 return this->_is_arg_name_used({arg.substr(this->_flag_prefix_char_length)});
554
555 return false;
556 }
557
562 void _strip_flag_prefix(std::string& arg_flag) const noexcept {
563 if (arg_flag.starts_with(this->_flag_prefix))
564 arg_flag.erase(0, this->_flag_prefix_length);
565 else
566 arg_flag.erase(0, this->_flag_prefix_char_length);
567 }

573 void _parse_args_impl(const arg_token_list_t& arg_tokens) {
574 arg_token_list_iterator_t token_it = arg_tokens.begin();
575 this->_parse_positional_args(arg_tokens, token_it);
576 this->_parse_optional_args(arg_tokens, token_it);
577 }
578
584 void _parse_positional_args(
585 const arg_token_list_t& arg_tokens, arg_token_list_iterator_t& token_it
586 ) noexcept {
587 for (const auto& pos_arg : this->_positional_args) {
588 if (token_it == arg_tokens.end())
589 return;
590
591 if (token_it->type == detail::argument_token::t_flag)
592 return;
593
594 pos_arg->set_value(token_it->value);
595 ++token_it;
596 }
597 }
598
606 void _parse_optional_args(
607 const arg_token_list_t& arg_tokens, arg_token_list_iterator_t& token_it
608 ) {
609 std::optional<std::reference_wrapper<arg_ptr_t>> curr_opt_arg;
610
611 while (token_it != arg_tokens.end()) {
612 if (token_it->type == detail::argument_token::t_flag) {
613 auto opt_arg_it = std::ranges::find_if(
614 this->_optional_args, this->_name_match_predicate(token_it->value)
615 );
616
617 if (opt_arg_it == this->_optional_args.end())
618 throw error::argument_not_found(token_it->value);
619
620 curr_opt_arg = std::ref(*opt_arg_it);
621 curr_opt_arg->get()->mark_used();
622 }
623 else {
624 if (not curr_opt_arg)
625 throw error::free_value(token_it->value);
626
627 curr_opt_arg->get()->set_value(token_it->value);
628 }
629
630 ++token_it;
631 }
632 }
633
638 [[nodiscard]] bool _are_required_args_bypassed() const noexcept {
639 return std::ranges::any_of(this->_optional_args, [](const arg_ptr_t& arg) {
640 return arg->is_used() and arg->bypass_required_enabled();
641 });
642 }
643
648 void _check_required_args() const {
649 for (const auto& arg : this->_positional_args)
650 if (not arg->is_used())
651 throw error::required_argument_not_parsed(arg->name());
652
653 for (const auto& arg : this->_optional_args)
654 if (arg->is_required() and not arg->has_value())
655 throw error::required_argument_not_parsed(arg->name());
656 }
657
662 void _check_nvalues_in_range() const {
663 for (const auto& arg : this->_positional_args) {
664 const auto nvalues_ordering = arg->nvalues_in_range();
665 if (not std::is_eq(nvalues_ordering))
666 throw error::invalid_nvalues(nvalues_ordering, arg->name());
667 }
668
669 for (const auto& arg : this->_optional_args) {
670 const auto nvalues_ordering = arg->nvalues_in_range();
671 if (not std::is_eq(nvalues_ordering))
672 throw error::invalid_nvalues(nvalues_ordering, arg->name());
673 }
674 }
675
681 arg_opt_t _get_argument(std::string_view arg_name) const noexcept {
682 const auto predicate = this->_name_match_predicate(arg_name);
683
684 if (auto pos_arg_it = std::ranges::find_if(this->_positional_args, predicate);
685 pos_arg_it != this->_positional_args.end()) {
686 return std::ref(**pos_arg_it);
687 }
688
689 if (auto opt_arg_it = std::ranges::find_if(this->_optional_args, predicate);
690 opt_arg_it != this->_optional_args.end()) {
691 return std::ref(**opt_arg_it);
692 }
693
694 return std::nullopt;
695 }

711inline void add_default_argument(
712 const argument::default_positional arg_discriminator, argument_parser& arg_parser
713) noexcept {
714 switch (arg_discriminator) {
715 case argument::default_positional::input:
716 arg_parser.add_positional_argument("input")
717 .action<action_type::modify>(action::check_file_exists())
718 .help("Input file path");
719 break;
720
721 case argument::default_positional::output:
722 arg_parser.add_positional_argument("output").help("Output file path");
723 break;
724 }
725}
726
727inline void add_default_argument(
728 const argument::default_optional arg_discriminator, argument_parser& arg_parser
729) noexcept {
730 switch (arg_discriminator) {
731 case argument::default_optional::help:
732 arg_parser.add_flag("help", "h").bypass_required().help("Display the help message");
733 break;
734
735 case argument::default_optional::input:
736 arg_parser.add_optional_argument("input", "i")
737 .required()
738 .nargs(1)
739 .action<action_type::modify>(action::check_file_exists())
740 .help("Input file path");
741 break;
742
743 case argument::default_optional::output:
744 arg_parser.add_optional_argument("output", "o").required().nargs(1).help("Output file path");
745 break;
746
747 case argument::default_optional::multi_input:
748 arg_parser.add_optional_argument("input", "i")
749 .required()
750 .nargs(ap::nargs::at_least(1))
751 .action<action_type::modify>(action::check_file_exists())
752 .help("Input files paths");
753 break;
754
755 case argument::default_optional::multi_output:
756 arg_parser.add_optional_argument("output", "o")
757 .required()
758 .nargs(ap::nargs::at_least(1))
759 .help("Output files paths");
760 break;
761 }
