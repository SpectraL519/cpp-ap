165 positional& set_value(const std::string& str_value) override {
166 if (this->_value.has_value())
167 throw error::value_already_set(this->_name);
168
169 this->_ss.clear();
170 this->_ss.str(str_value);
171
172 value_type value;
173 if (not (this->_ss >> value))
174 throw error::invalid_value(this->_name, this->_ss.str());
175
176 if (not this->_is_valid_choice(value))
177 throw error::invalid_choice(this->_name, str_value);
178
179 this->_apply_action(value);
180
181 this->_value = value;
182 return *this;
183 }

242 }
243 else {

246 value = std::get<callable_type>(this->_action)(value);
247 }
