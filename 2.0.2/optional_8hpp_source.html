236 optional& set_value(const std::string& str_value) override {
237 if (not (this->_nargs_range or this->_values.empty()))
238 throw error::value_already_set(this->_name);
239
240 this->_ss.clear();
241 this->_ss.str(str_value);
242
243 value_type value;
244 if (not (this->_ss >> value))
245 throw error::invalid_value(this->_name, this->_ss.str());
246
247 if (not this->_is_valid_choice(value))
248 throw error::invalid_choice(this->_name, str_value);
249
250 this->_apply_action(value);
251
252 this->_values.emplace_back(std::move(value));
253 return *this;
254 }

257 [[nodiscard]] bool has_value() const noexcept override {

267 [[nodiscard]] std::weak_ordering nvalues_in_range() const noexcept override {

297 [[nodiscard]] const std::any& _predefined_value() const {
298 if (this->is_used()) {
299 if (not this->_implicit_value.has_value())
300 throw(std::logic_error(
301 std::format("No implicit value specified for argument `{}`.", this->_name.str())
302 ));
303

305 }
306

308 throw(std::logic_error(
309 std::format("No default value specified for argument `{}`.", this->_name.str())
310 ));
311
312 return this->_default_value;
313 }

320 [[nodiscard]] bool _is_valid_choice(const value_type& choice) const noexcept {
321 // TODO: replace with `std::ranges::contains` after transition to C++23
322 return this->_choices.empty()
323 or std::ranges::find(this->_choices, choice) != this->_choices.end();
324 }

330 void _apply_action(value_type& value) const noexcept {
331 if (action::detail::is_modify_action(this->_action)) {
332 using callable_type =
333 action::detail::callable_type<ap::action_type::modify, value_type>;
334 std::get<callable_type>(this->_action)(value);
335 }
336 else {
337 using callable_type =
338 action::detail::callable_type<ap::action_type::transform, value_type>;
339 value = std::get<callable_type>(this->_action)(value);
340 }
341 }
