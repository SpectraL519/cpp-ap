113struct argument_name {
114 argument_name() = delete;
115 argument_name& operator=(const argument_name&) = delete;
116 argument_name& operator=(argument_name&&) = delete;
117
119 argument_name(const argument_name&) = default;
120
122 argument_name(argument_name&&) = default;
123
128 argument_name(std::string_view primary) : primary(primary) {}
129
135 argument_name(std::string_view primary, std::string_view secondary)
136 : primary(primary), secondary(secondary) {}
137
139 ~argument_name() = default;
140
146 bool operator==(const argument_name& other) const noexcept {
147 if (not (this->secondary and other.secondary) and (this->secondary or other.secondary))
148 return false;
149
150 if (this->primary != other.primary)
151 return false;
152
153 return this->secondary ? this->secondary.value() == other.secondary.value() : true;
154 }
155
161 [[nodiscard]] bool match(std::string_view arg_name) const noexcept {
162 return arg_name == this->primary or (this->secondary and arg_name == this->secondary.value());
163 }
164
170 [[nodiscard]] bool match(const argument_name& arg_name) const noexcept {
171 if (this->match(arg_name.primary))
172 return true;
173
174 if (arg_name.secondary)
175 return this->match(arg_name.secondary.value());
176
177 return false;
178 }
179
181 [[nodiscard]] std::string str() const noexcept {
182 return this->secondary
183 ? ("[" + this->primary + "," + this->secondary.value() + "]")
184 : ("[" + this->primary + "]");
185 }
186
193 friend std::ostream& operator<<(std::ostream& os, const argument_name& arg_name) noexcept {
194 os << arg_name.str();
195 return os;
196 }
197
198 const std::string primary;
199 const std::optional<std::string> secondary;
200};

469 [[nodiscard]] std::weak_ordering contains(const range::count_type n) const noexcept {
470 if (not (this->_nlow.has_value() or this->_nhigh.has_value()))
471 return std::weak_ordering::equivalent;
472
473 if (this->_nlow.has_value() and this->_nhigh.has_value()) {
474 if (n < this->_nlow.value())
475 return std::weak_ordering::less;
476
477 if (n > this->_nhigh.value())
478 return std::weak_ordering::greater;
479
480 return std::weak_ordering::equivalent;
481 }
482
483 if (this->_nlow.has_value())
484 return (n < this->_nlow.value()) ? std::weak_ordering::less : std::weak_ordering::equivalent;
485
486 return (n > this->_nhigh.value()) ? std::weak_ordering::greater : std::weak_ordering::equivalent;
487 }
