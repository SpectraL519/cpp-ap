171 bool set_value(const std::string& str_value) override {
172 if (this->_value.has_value())
173 throw parsing_failure::value_already_set(this->_name);
174
175 value_type value;
176 if (not (std::istringstream(str_value) >> value))
177 throw parsing_failure::invalid_value(this->_name, str_value);
178
179 if (not detail::is_valid_choice(value, this->_choices))
180 throw parsing_failure::invalid_choice(this->_name, str_value);
181
182 const auto apply_visitor = action::detail::apply_visitor<value_type>{value};
183 for (const auto& action : this->_value_actions)
184 std::visit(apply_visitor, action);
185
186 this->_value = value;
187 return false;
188 }
