213 if (not verbose)

216 desc.params.reserve(6);
217 if (this->_required)

220 desc.add_param("bypass required", "true");
221 if (this->_nargs_range.has_value())
222 desc.add_param("nargs", this->_nargs_range.value());
223 if constexpr (detail::c_writable<value_type>) {

268 optional& set_value(const std::string& str_value) override {
269 if (not (this->_nargs_range or this->_values.empty()))
270 throw error::value_already_set(this->_name);
271
272 this->_ss.clear();
273 this->_ss.str(str_value);
274
275 value_type value;
276 if (not (this->_ss >> value))
277 throw error::invalid_value(this->_name, this->_ss.str());
278
279 if (not this->_is_valid_choice(value))
280 throw error::invalid_choice(this->_name, str_value);
281
282 this->_apply_action(value);
283
284 this->_values.emplace_back(std::move(value));
285 return *this;
286 }

362 void _apply_action(value_type& value) const noexcept {
363 if (action::detail::is_modify_action(this->_action)) {
364 using callable_type =
365 action::detail::callable_type<ap::action_type::modify, value_type>;
366 std::get<callable_type>(this->_action)(value);
367 }
368 else {
369 using callable_type =
370 action::detail::callable_type<ap::action_type::transform, value_type>;
371 value = std::get<callable_type>(this->_action)(value);
372 }
373 }
