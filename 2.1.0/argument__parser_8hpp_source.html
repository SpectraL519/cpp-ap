358 template <detail::c_argument_value_type T = std::string>
359 T value(std::string_view arg_name) const {
360 const auto arg_opt = this->_get_argument(arg_name);
361 if (not arg_opt)
362 throw error::argument_not_found(arg_name);

365 try {

382 template <detail::c_argument_value_type T = std::string, std::convertible_to<T> U>
383 T value_or(std::string_view arg_name, U&& default_value) const {
384 const auto arg_opt = this->_get_argument(arg_name);
385 if (not arg_opt)
386 throw error::argument_not_found(arg_name);
387
388 try {
389 const auto& arg_value = arg_opt->get().value();
390 return std::any_cast<T>(arg_value);
391 }
392 catch (const std::logic_error&) {
393 // positional: no value parsed
394 // optional: no value parsed + no predefined value
395 return T{std::forward<U>(default_value)};
396 }
397 catch (const std::bad_any_cast& err) {
398 throw error::invalid_value_type(arg_opt->get().name(), typeid(T));
399 }

409 template <detail::c_argument_value_type T = std::string>
410 std::vector<T> values(std::string_view arg_name) const {
411 const auto arg_opt = this->_get_argument(arg_name);
412 if (not arg_opt)
413 throw error::argument_not_found(arg_name);
414
415 const auto& arg = arg_opt->get();
416
417 try {
418 if (not arg.has_parsed_values() and arg.has_value())
419 return std::vector<T>{std::any_cast<T>(arg.value())};
420
421 std::vector<T> values;
422 // TODO: use std::ranges::to after transition to C++23
423 std::ranges::copy(
424 std::views::transform(
425 arg.values(), [](const std::any& value) { return std::any_cast<T>(value); }
426 ),
427 std::back_inserter(values)
428 );
429 return values;
430 }
431 catch (const std::bad_any_cast& err) {
432 throw error::invalid_value_type(arg.name(), typeid(T));
433 }

493 void _verify_arg_name_pattern(const std::string_view arg_name) const {
494 if (arg_name.empty())
495 throw error::invalid_argument_name_pattern(
496 arg_name, "An argument name cannot be empty."
497 );
498
499 if (arg_name.front() == this->_flag_prefix_char)
500 throw error::invalid_argument_name_pattern(
501 arg_name,
502 std::format(
503 "An argument name cannot begin with a flag prefix character ({}).",
504 this->_flag_prefix_char
505 )
506 );
507
508 if (std::isdigit(arg_name.front()))
509 throw error::invalid_argument_name_pattern(
510 arg_name, "An argument name cannot begin with a digit."
511 );
512 }

519 [[nodiscard]] auto _name_match_predicate(const std::string_view arg_name) const noexcept {
520 return [arg_name](const arg_ptr_t& arg) { return arg->name().match(arg_name); };
521 }
522
528 [[nodiscard]] auto _name_match_predicate(const detail::argument_name& arg_name) const noexcept {
529 return [&arg_name](const arg_ptr_t& arg) { return arg->name().match(arg_name); };
530 }

537 [[nodiscard]] bool _is_arg_name_used(const detail::argument_name& arg_name) const noexcept {
538 const auto predicate = this->_name_match_predicate(arg_name);
539
540 if (std::ranges::find_if(this->_positional_args, predicate) != this->_positional_args.end())
541 return true;
542
543 if (std::ranges::find_if(this->_optional_args, predicate) != this->_optional_args.end())
544 return true;
545
546 return false;

555 template <detail::c_sized_range_of<std::string, detail::type_validator::convertible> AR>
556 [[nodiscard]] arg_token_list_t _tokenize(const AR& arg_range) const noexcept {
557 const auto n_args = std::ranges::size(arg_range);
558 if (n_args == 0ull)
559 return arg_token_list_t{};
560
561 arg_token_list_t toks;
562 toks.reserve(n_args);
563
564 for (const auto& arg : arg_range) {
565 std::string value = static_cast<std::string>(arg);
566 if (this->_is_flag(value)) {
567 this->_strip_flag_prefix(value);
568 toks.emplace_back(detail::argument_token::t_flag, std::move(value));
569 }
570 else {
571 toks.emplace_back(detail::argument_token::t_value, std::move(value));
572 }
573 }
574
575 return toks;
576 }
577
583 [[nodiscard]] bool _is_flag(const std::string& arg) const noexcept {
584 if (arg.starts_with(this->_flag_prefix))
585 return this->_is_arg_name_used({arg.substr(this->_primary_flag_prefxi_length)});
586
587 if (arg.starts_with(this->_flag_prefix_char))
588 return this->_is_arg_name_used({arg.substr(this->_secondary_flag_prefix_length)});
589
590 return false;
591 }
592
597 void _strip_flag_prefix(std::string& arg_flag) const noexcept {
598 if (arg_flag.starts_with(this->_flag_prefix))
599 arg_flag.erase(0, this->_primary_flag_prefxi_length);
600 else
601 arg_flag.erase(0, this->_secondary_flag_prefix_length);
602 }

608 void _parse_args_impl(const arg_token_list_t& arg_tokens) {
609 arg_token_list_iterator_t token_it = arg_tokens.begin();
610 this->_parse_positional_args(arg_tokens, token_it);
611 this->_parse_optional_args(arg_tokens, token_it);
612 }
613
619 void _parse_positional_args(
620 const arg_token_list_t& arg_tokens, arg_token_list_iterator_t& token_it
621 ) noexcept {
622 for (const auto& pos_arg : this->_positional_args) {
623 if (token_it == arg_tokens.end())
624 return;
625
626 if (token_it->type == detail::argument_token::t_flag)
627 return;
628
629 pos_arg->set_value(token_it->value);
630 ++token_it;
631 }
632 }
633
641 void _parse_optional_args(
642 const arg_token_list_t& arg_tokens, arg_token_list_iterator_t& token_it
643 ) {
644 std::optional<std::reference_wrapper<arg_ptr_t>> curr_opt_arg;
645
646 while (token_it != arg_tokens.end()) {
647 if (token_it->type == detail::argument_token::t_flag) {
648 auto opt_arg_it = std::ranges::find_if(
649 this->_optional_args, this->_name_match_predicate(token_it->value)
650 );
651
652 if (opt_arg_it == this->_optional_args.end())
653 throw error::argument_not_found(token_it->value);
654
655 curr_opt_arg = std::ref(*opt_arg_it);
656 curr_opt_arg->get()->mark_used();
657 }
658 else {
659 if (not curr_opt_arg)
660 throw error::free_value(token_it->value);
661
662 curr_opt_arg->get()->set_value(token_it->value);
663 }
664
665 ++token_it;
666 }
667 }

673 [[nodiscard]] bool _are_required_args_bypassed() const noexcept {
674 return std::ranges::any_of(this->_optional_args, [](const arg_ptr_t& arg) {
675 return arg->is_used() and arg->bypass_required_enabled();
676 });
677 }
678
683 void _check_required_args() const {
684 for (const auto& arg : this->_positional_args)
685 if (not arg->is_used())
686 throw error::required_argument_not_parsed(arg->name());
687
688 for (const auto& arg : this->_optional_args)
689 if (arg->is_required() and not arg->has_value())
690 throw error::required_argument_not_parsed(arg->name());
691 }
692
697 void _check_nvalues_in_range() const {
698 for (const auto& arg : this->_positional_args) {
699 const auto nvalues_ordering = arg->nvalues_in_range();
700 if (not std::is_eq(nvalues_ordering))
701 throw error::invalid_nvalues(nvalues_ordering, arg->name());
702 }
703
704 for (const auto& arg : this->_optional_args) {
705 const auto nvalues_ordering = arg->nvalues_in_range();
706 if (not std::is_eq(nvalues_ordering))
707 throw error::invalid_nvalues(nvalues_ordering, arg->name());
708 }
709 }
710
716 arg_opt_t _get_argument(std::string_view arg_name) const noexcept {
717 const auto predicate = this->_name_match_predicate(arg_name);
718
719 if (auto pos_arg_it = std::ranges::find_if(this->_positional_args, predicate);
720 pos_arg_it != this->_positional_args.end()) {
721 return std::ref(**pos_arg_it);
722 }
723
724 if (auto opt_arg_it = std::ranges::find_if(this->_optional_args, predicate);
725 opt_arg_it != this->_optional_args.end()) {
726 return std::ref(**opt_arg_it);
727 }
728
729 return std::nullopt;
730 }

737 void _print(std::ostream& os, const arg_ptr_list_t& args) const noexcept {
738 if (this->_verbose) {
739 for (const auto& arg : args)
740 os << '\n' << arg->desc(this->_verbose).get(this->_indent_width) << '\n';
741 }
742 else {
743 std::vector<detail::argument_descriptor> descriptors;
744 descriptors.reserve(args.size());
745
746 for (const auto& arg : args)
747 descriptors.emplace_back(arg->desc(this->_verbose));
748
749 std::size_t max_arg_name_length = 0ull;
750 for (const auto& desc : descriptors)
751 max_arg_name_length = std::max(max_arg_name_length, desc.name.length());
752
753 for (const auto& desc : descriptors)
754 os << '\n' << desc.get_basic(this->_indent_width, max_arg_name_length);
755
756 os << '\n';
757 }
758 }

781inline void add_default_argument(
782 const argument::default_positional arg_discriminator, argument_parser& arg_parser
783) noexcept {
784 switch (arg_discriminator) {
785 case argument::default_positional::input:
786 arg_parser.add_positional_argument("input")
787 .action<action_type::modify>(action::check_file_exists())
788 .help("Input file path");
789 break;
790
791 case argument::default_positional::output:
792 arg_parser.add_positional_argument("output").help("Output file path");
793 break;
794 }
795}
796
802inline void add_default_argument(
803 const argument::default_optional arg_discriminator, argument_parser& arg_parser
804) noexcept {
805 switch (arg_discriminator) {
806 case argument::default_optional::help:
807 arg_parser.add_flag("help", "h").bypass_required().help("Display the help message");
808 break;
809
810 case argument::default_optional::input:
811 arg_parser.add_optional_argument("input", "i")
812 .required()
813 .nargs(1)
814 .action<action_type::modify>(action::check_file_exists())
815 .help("Input file path");
816 break;
817
818 case argument::default_optional::output:
819 arg_parser.add_optional_argument("output", "o").required().nargs(1).help("Output file path");
820 break;
821
822 case argument::default_optional::multi_input:
823 arg_parser.add_optional_argument("input", "i")
824 .required()
825 .nargs(ap::nargs::at_least(1))
826 .action<action_type::modify>(action::check_file_exists())
827 .help("Input files paths");
828 break;
829
830 case argument::default_optional::multi_output:
831 arg_parser.add_optional_argument("output", "o")
832 .required()
833 .nargs(ap::nargs::at_least(1))
834 .help("Output files paths");
835 break;
836 }
