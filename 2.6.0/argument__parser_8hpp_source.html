400 template <detail::c_argument_value_type T = std::string>
401 [[nodiscard]] T value(std::string_view arg_name) const {
402 const auto arg_opt = this->_get_argument(arg_name);
403 if (not arg_opt)
404 throw lookup_failure::argument_not_found(arg_name);
405
406 const auto& arg_value = arg_opt->get().value();
407 try {
408 return std::any_cast<T>(arg_value);
409 }
410 catch (const std::bad_any_cast&) {
411 throw type_error::invalid_value_type<T>(arg_opt->get().name());
412 }

423 template <detail::c_argument_value_type T = std::string, std::convertible_to<T> U>
424 [[nodiscard]] T value_or(std::string_view arg_name, U&& fallback_value) const {
425 const auto arg_opt = this->_get_argument(arg_name);
426 if (not arg_opt)
427 throw lookup_failure::argument_not_found(arg_name);
428
429 try {
430 const auto& arg_value = arg_opt->get().value();
431 return std::any_cast<T>(arg_value);
432 }
433 catch (const std::logic_error&) {
434 // positional: no value parsed
435 // optional: no value parsed + no predefined value
436 return T{std::forward<U>(fallback_value)};
437 }
438 catch (const std::bad_any_cast&) {
439 throw type_error::invalid_value_type<T>(arg_opt->get().name());
440 }
441 }

449 template <detail::c_argument_value_type T = std::string>
450 [[nodiscard]] std::vector<T> values(std::string_view arg_name) const {
451 const auto arg_opt = this->_get_argument(arg_name);
452 if (not arg_opt)
453 throw lookup_failure::argument_not_found(arg_name);
454
455 const auto& arg = arg_opt->get();
456
457 try {
458 if (not arg.has_parsed_values() and arg.has_value())
459 return std::vector<T>{std::any_cast<T>(arg.value())};
460
461 std::vector<T> values;
462 // TODO: use std::ranges::to after transition to C++23
463 std::ranges::copy(
464 std::views::transform(
465 arg.values(), [](const std::any& value) { return std::any_cast<T>(value); }
466 ),
467 std::back_inserter(values)
468 );
469 return values;
470 }
471 catch (const std::bad_any_cast&) {
472 throw type_error::invalid_value_type<T>(arg.name());
473 }

540 void _verify_arg_name_pattern(const std::string_view arg_name) const {
541 if (arg_name.empty())
542 throw invalid_configuration::invalid_argument_name(
543 arg_name, "An argument name cannot be empty."
544 );
545
546 if (detail::contains_whitespaces(arg_name))
547 throw invalid_configuration::invalid_argument_name(
548 arg_name, "An argument name cannot contain whitespaces."
549 );
550
551 if (arg_name.front() == this->_flag_prefix_char)
552 throw invalid_configuration::invalid_argument_name(
553 arg_name,
554 std::format(
555 "An argument name cannot begin with a flag prefix character ({}).",
556 this->_flag_prefix_char
557 )
558 );
559
560 if (std::isdigit(arg_name.front()))
561 throw invalid_configuration::invalid_argument_name(
562 arg_name, "An argument name cannot begin with a digit."
563 );
564 }

621 void _validate_argument_configuration() const {
622 // step 1
623 const_arg_opt_t non_required_arg = std::nullopt;
624 for (const auto& arg : this->_positional_args) {
625 if (not arg->is_required()) {
626 non_required_arg = std::ref(*arg);
627 continue;
628 }
629
630 if (non_required_arg and arg->is_required())
631 throw invalid_configuration::positional::required_after_non_required(
632 arg->name(), non_required_arg->get().name()
633 );
634 }

643 template <detail::c_sized_range_of<std::string_view, detail::type_validator::convertible> AR>
644 [[nodiscard]] arg_token_list_t _tokenize(const AR& arg_range) {
645 const auto n_args = std::ranges::size(arg_range);
646 if (n_args == 0ull)
647 return arg_token_list_t{};
648
649 arg_token_list_t toks;
650 toks.reserve(n_args);
651 std::ranges::for_each(
652 arg_range, std::bind_front(&argument_parser::_tokenize_arg, this, std::ref(toks))
653 );
654 return toks;
655 }
656
662 void _tokenize_arg(arg_token_list_t& toks, const std::string_view arg_value) {
663 auto tok = this->_build_token(arg_value);
664
665 if (not tok.is_flag_token() or this->_validate_flag_token(tok)) {
666 toks.emplace_back(std::move(tok));
667 return;
668 }
669
670 // invalid flag - check for compound secondary flag
671 const auto compound_toks = this->_try_split_compound_flag(tok);
672 if (compound_toks.empty()) { // not a valid compound flag
673#ifdef AP_UNKNOWN_FLAGS_AS_VALUES
674 toks.emplace_back(detail::argument_token::t_value, std::string(arg_value));
675 return;
676#else
677 throw parsing_failure::unknown_argument(arg_value);
678#endif
679 }
680
681 toks.insert(toks.end(), compound_toks.begin(), compound_toks.end());
682 }

689 [[nodiscard]] detail::argument_token _build_token(const std::string_view arg_value
690 ) const noexcept {
691 if (detail::contains_whitespaces(arg_value))
692 return {.type = detail::argument_token::t_value, .value = std::string(arg_value)};
693
694 if (arg_value.starts_with(this->_flag_prefix))
695 return {
696 .type = detail::argument_token::t_flag_primary,
697 .value = std::string(arg_value.substr(this->_primary_flag_prefix_length))
698 };
699
700 if (arg_value.starts_with(this->_flag_prefix_char))
701 return {
702 .type = detail::argument_token::t_flag_secondary,
703 .value = std::string(arg_value.substr(this->_secondary_flag_prefix_length))
704 };
705
706 return {.type = detail::argument_token::t_value, .value = std::string(arg_value)};
707 }

715 [[nodiscard]] bool _validate_flag_token(detail::argument_token& tok) noexcept {
716 const auto opt_arg_it = this->_find_opt_arg(tok);
717 if (opt_arg_it == this->_optional_args.end())
718 return false;
719
720 tok.arg.emplace(*opt_arg_it);
721 return true;
722 }

730 [[nodiscard]] std::vector<detail::argument_token> _try_split_compound_flag(
731 const detail::argument_token& tok
732 ) noexcept {
733 std::vector<detail::argument_token> compound_toks;
734 compound_toks.reserve(tok.value.size());
735
736 if (tok.type != detail::argument_token::t_flag_secondary)
737 return compound_toks;
738
739 for (const char c : tok.value) {
740 detail::argument_token ctok{
741 detail::argument_token::t_flag_secondary, std::string(1ull, c)
742 };
743 if (not this->_validate_flag_token(ctok)) {
744 compound_toks.clear();
745 return compound_toks;
746 }
747 compound_toks.emplace_back(std::move(ctok));
748 }
749
750 return compound_toks;
751 }

783 void _parse_args_impl(const arg_token_list_t& arg_tokens) {
784 arg_token_list_iterator_t token_it = arg_tokens.begin();
785
786 this->_parse_positional_args(token_it, arg_tokens.end());
787
788 std::vector<std::string_view> unknown_args;
789 this->_parse_optional_args(token_it, arg_tokens.end(), unknown_args);
790
791 if (not unknown_args.empty())
792 throw parsing_failure::argument_deduction_failure(unknown_args);
793 }
794
800 void _parse_positional_args(
801 arg_token_list_iterator_t& token_it, const arg_token_list_iterator_t& tokens_end
802 ) noexcept {
803 for (const auto& pos_arg : this->_positional_args) {
804 if (token_it == tokens_end)
805 return;
806
807 if (token_it->type != detail::argument_token::t_value)
808 return;
809
810 pos_arg->set_value(token_it->value);
811 ++token_it;
812 }
813 }
814
823 void _parse_optional_args(
824 arg_token_list_iterator_t& token_it,
825 const arg_token_list_iterator_t& tokens_end,
826 std::vector<std::string_view>& unknown_args
827 ) {
828 arg_ptr_opt_t curr_opt_arg;
829
830 while (token_it != tokens_end) {
831 switch (token_it->type) {
832 case detail::argument_token::t_flag_primary:
833 [[fallthrough]];
834 case detail::argument_token::t_flag_secondary: {
835 curr_opt_arg = token_it->arg;
836 if (not curr_opt_arg->get()->mark_used())
837 curr_opt_arg.reset();
838
839 break;
840 }
841 case detail::argument_token::t_value: {
842 if (not curr_opt_arg) {
843 unknown_args.emplace_back(token_it->value);
844 break;
845 }
846
847 if (not curr_opt_arg->get()->set_value(token_it->value))
848 curr_opt_arg.reset();
849
850 break;
851 }
852 }
853
854 ++token_it;
855 }
856 }

862 [[nodiscard]] bool _are_required_args_bypassed() const noexcept {
863 // TODO: use std::views::join after the transition to C++23
864 return std::ranges::any_of(
865 this->_positional_args,
866 [](const arg_ptr_t& arg) {
867 return arg->is_used() and arg->bypass_required_enabled();
868 }
869 )
870 or std::ranges::any_of(this->_optional_args, [](const arg_ptr_t& arg) {
871 return arg->is_used() and arg->bypass_required_enabled();
872 });
873 }
874
879 void _verify_required_args() const {
880 // TODO: use std::views::join after the transition to C++23
881 for (const auto& arg : this->_positional_args)
882 if (arg->is_required() and not arg->has_value())
883 throw parsing_failure::required_argument_not_parsed(arg->name());
884
885 for (const auto& arg : this->_optional_args)
886 if (arg->is_required() and not arg->has_value())
887 throw parsing_failure::required_argument_not_parsed(arg->name());
888 }
889
894 void _verify_nvalues() const {
895 // TODO: use std::views::join after the transition to C++23
896 for (const auto& arg : this->_positional_args)
897 if (const auto nv_ord = arg->nvalues_ordering(); not std::is_eq(nv_ord))
898 throw parsing_failure::invalid_nvalues(arg->name(), nv_ord);
899
900 for (const auto& arg : this->_optional_args)
901 if (const auto nv_ord = arg->nvalues_ordering(); not std::is_eq(nv_ord))
902 throw parsing_failure::invalid_nvalues(arg->name(), nv_ord);
903 }

910 arg_opt_t _get_argument(std::string_view arg_name) const noexcept {
911 const auto predicate = this->_name_match_predicate(arg_name);
912
913 if (auto pos_arg_it = std::ranges::find_if(this->_positional_args, predicate);
914 pos_arg_it != this->_positional_args.end()) {
915 return std::ref(**pos_arg_it);
916 }
917
918 if (auto opt_arg_it = std::ranges::find_if(this->_optional_args, predicate);
919 opt_arg_it != this->_optional_args.end()) {
920 return std::ref(**opt_arg_it);
921 }
922
923 return std::nullopt;
924 }

932 [[nodiscard]] arg_ptr_list_iter_t _find_opt_arg(const detail::argument_token& flag_tok
933 ) noexcept {
934 switch (flag_tok.type) {
935 case detail::argument_token::t_flag_primary:
936 return std::ranges::find(this->_optional_args, flag_tok.value, [](const auto& arg_ptr) {
937 return arg_ptr->name().primary;
938 });
939 case detail::argument_token::t_flag_secondary:
940 return std::ranges::find(this->_optional_args, flag_tok.value, [](const auto& arg_ptr) {
941 return arg_ptr->name().secondary;
942 });
943 default:
944 return this->_optional_args.end();
945 }

953 void _print(std::ostream& os, const arg_ptr_list_t& args, const bool verbose) const noexcept {
954 auto visible_args =
955 std::views::filter(args, [](const auto& arg) { return not arg->is_hidden(); });
956
957 if (verbose) {
