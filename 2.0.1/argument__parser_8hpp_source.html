321 template <detail::c_argument_value_type T = std::string>
322 T value(std::string_view arg_name) const {
323 const auto arg_opt = this->_get_argument(arg_name);
324 if (not arg_opt)
325 throw error::argument_not_found(arg_name);
326
327 const auto& arg_value = arg_opt->get().value();
328 try {
329 return std::any_cast<T>(arg_value);
330 }
331 catch (const std::bad_any_cast& err) {
332 throw error::invalid_value_type(arg_opt->get().name(), typeid(T));
333 }
334 }

343 template <detail::c_argument_value_type T = std::string, std::convertible_to<T> U>
344 T value_or(std::string_view arg_name, U&& default_value) const {
345 const auto arg_opt = this->_get_argument(arg_name);
346 if (not arg_opt)
347 throw error::argument_not_found(arg_name);
348
349 try {
350 const auto& arg_value = arg_opt->get().value();
351 return std::any_cast<T>(arg_value);
352 }
353 catch (const std::logic_error&) {
354 // positional: no value parsed
355 // optional: no value parsed + no predefined value
356 return T{std::forward<U>(default_value)};
357 }
358 catch (const std::bad_any_cast& err) {
359 throw error::invalid_value_type(arg_opt->get().name(), typeid(T));
360 }
361 }

368 template <detail::c_argument_value_type T = std::string>
369 std::vector<T> values(std::string_view arg_name) const {
370 const auto arg_opt = this->_get_argument(arg_name);
371 if (not arg_opt)
372 throw error::argument_not_found(arg_name);
373
374 const auto& arg = arg_opt->get();
375
376 try {
377 if (not arg.has_parsed_values() and arg.has_value())
378 return std::vector<T>{std::any_cast<T>(arg.value())};
379
380 std::vector<T> values;
381 // TODO: use std::ranges::to after transition to C++23
382 std::ranges::copy(
383 std::views::transform(
384 arg.values(), [](const std::any& value) { return std::any_cast<T>(value); }
385 ),
386 std::back_inserter(values)
387 );
388 return values;
389 }
390 catch (const std::bad_any_cast& err) {
391 throw error::invalid_value_type(arg.name(), typeid(T));
392 }

453 [[nodiscard]] bool _is_arg_name_used(const detail::argument_name& arg_name) const noexcept {
454 const auto predicate = this->_name_match_predicate(arg_name);
455
456 if (std::ranges::find_if(this->_positional_args, predicate) != this->_positional_args.end())

471 template <detail::c_sized_range_of<std::string, detail::type_validator::convertible> AR>
472 [[nodiscard]] arg_token_list_t _tokenize(const AR& arg_range) const noexcept {
473 const auto n_args = std::ranges::size(arg_range);
474 if (n_args == 0ull)
475 return arg_token_list_t{};
476
477 arg_token_list_t toks;
478 toks.reserve(n_args);
479
480 for (const auto& arg : arg_range) {
481 std::string value = static_cast<std::string>(arg);
482 if (this->_is_flag(value)) {
483 this->_strip_flag_prefix(value);
484 toks.emplace_back(detail::argument_token::t_flag, std::move(value));
485 }
486 else {
487 toks.emplace_back(detail::argument_token::t_value, std::move(value));
488 }
489 }
490
491 return toks;
492 }
493
499 [[nodiscard]] bool _is_flag(const std::string& arg) const noexcept {
500 if (arg.starts_with(this->_flag_prefix))
501 return this->_is_arg_name_used({arg.substr(this->_flag_prefix_length)});
502
503 if (arg.starts_with(this->_flag_prefix_char))
504 return this->_is_arg_name_used({arg.substr(this->_flag_prefix_char_length)});
505
506 return false;
507 }
508
513 void _strip_flag_prefix(std::string& arg_flag) const noexcept {
514 if (arg_flag.starts_with(this->_flag_prefix))
515 arg_flag.erase(0, this->_flag_prefix_length);
516 else
517 arg_flag.erase(0, this->_flag_prefix_char_length);
518 }

524 void _parse_args_impl(const arg_token_list_t& arg_tokens) {
525 arg_token_list_iterator_t token_it = arg_tokens.begin();
526 this->_parse_positional_args(arg_tokens, token_it);
527 this->_parse_optional_args(arg_tokens, token_it);
528 }
529
535 void _parse_positional_args(
536 const arg_token_list_t& arg_tokens, arg_token_list_iterator_t& token_it
537 ) noexcept {
538 for (const auto& pos_arg : this->_positional_args) {
539 if (token_it == arg_tokens.end())
540 return;
541
542 if (token_it->type == detail::argument_token::t_flag)
543 return;
544
545 pos_arg->set_value(token_it->value);
546 ++token_it;
547 }
548 }
549
555 void _parse_optional_args(
556 const arg_token_list_t& arg_tokens, arg_token_list_iterator_t& token_it
557 ) {
558 std::optional<std::reference_wrapper<arg_ptr_t>> curr_opt_arg;
559
560 while (token_it != arg_tokens.end()) {
561 if (token_it->type == detail::argument_token::t_flag) {
562 auto opt_arg_it = std::ranges::find_if(
563 this->_optional_args, this->_name_match_predicate(token_it->value)
564 );
565
566 if (opt_arg_it == this->_optional_args.end())
567 throw error::argument_not_found(token_it->value);
568
569 curr_opt_arg = std::ref(*opt_arg_it);
570 curr_opt_arg->get()->mark_used();
571 }
572 else {
573 if (not curr_opt_arg)
574 throw error::free_value(token_it->value);
575
576 curr_opt_arg->get()->set_value(token_it->value);
577 }
578
579 ++token_it;
580 }
581 }

587 [[nodiscard]] bool _are_required_args_bypassed() const noexcept {
588 return std::ranges::any_of(this->_optional_args, [](const arg_ptr_t& arg) {
589 return arg->is_used() and arg->bypass_required_enabled();
590 });
591 }
592
594 void _check_required_args() const {
595 for (const auto& arg : this->_positional_args)
596 if (not arg->is_used())
597 throw error::required_argument_not_parsed(arg->name());
598
599 for (const auto& arg : this->_optional_args)
600 if (arg->is_required() and not arg->has_value())
601 throw error::required_argument_not_parsed(arg->name());
602 }
603
605 void _check_nvalues_in_range() const {
606 for (const auto& arg : this->_positional_args) {
607 const auto nvalues_ordering = arg->nvalues_in_range();
608 if (not std::is_eq(nvalues_ordering))
609 throw error::invalid_nvalues(nvalues_ordering, arg->name());
610 }
611
612 for (const auto& arg : this->_optional_args) {
613 const auto nvalues_ordering = arg->nvalues_in_range();
614 if (not std::is_eq(nvalues_ordering))
615 throw error::invalid_nvalues(nvalues_ordering, arg->name());
616 }
617 }
618
624 arg_opt_t _get_argument(std::string_view arg_name) const noexcept {
625 const auto predicate = this->_name_match_predicate(arg_name);
626
627 if (auto pos_arg_it = std::ranges::find_if(this->_positional_args, predicate);
628 pos_arg_it != this->_positional_args.end()) {
629 return std::ref(**pos_arg_it);
630 }
631
632 if (auto opt_arg_it = std::ranges::find_if(this->_optional_args, predicate);
633 opt_arg_it != this->_optional_args.end()) {
634 return std::ref(**opt_arg_it);
635 }
636
637 return std::nullopt;
638 }

654inline void add_default_argument(
655 const argument::default_positional arg_discriminator, argument_parser& arg_parser
656) noexcept {
657 switch (arg_discriminator) {
658 case argument::default_positional::input:
659 arg_parser.add_positional_argument("input")
660 .action<action_type::modify>(action::check_file_exists())
661 .help("Input file path");
662 break;
663
664 case argument::default_positional::output:
665 arg_parser.add_positional_argument("output").help("Output file path");
666 break;
667 }
668}
669
670inline void add_default_argument(
671 const argument::default_optional arg_discriminator, argument_parser& arg_parser
672) noexcept {
673 switch (arg_discriminator) {
674 case argument::default_optional::help:
675 arg_parser.add_flag("help", "h").bypass_required().help("Display the help message");
676 break;
677
678 case argument::default_optional::input:
679 arg_parser.add_optional_argument("input", "i")
680 .required()
681 .nargs(1)
682 .action<action_type::modify>(action::check_file_exists())
683 .help("Input file path");
684 break;
685
686 case argument::default_optional::output:
687 arg_parser.add_optional_argument("output", "o").required().nargs(1).help("Output file path");
688 break;
689
690 case argument::default_optional::multi_input:
691 arg_parser.add_optional_argument("input", "i")
692 .required()
693 .nargs(ap::nargs::at_least(1))
694 .action<action_type::modify>(action::check_file_exists())
695 .help("Input files paths");
696 break;
697
698 case argument::default_optional::multi_output:
699 arg_parser.add_optional_argument("output", "o")
700 .required()
701 .nargs(ap::nargs::at_least(1))
702 .help("Output files paths");
703 break;
704 }
