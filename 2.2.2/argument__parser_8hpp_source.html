362 template <detail::c_argument_value_type T = std::string>
363 [[nodiscard]] T value(std::string_view arg_name) const {
364 const auto arg_opt = this->_get_argument(arg_name);
365 if (not arg_opt)
366 throw error::argument_not_found(arg_name);
367
368 const auto& arg_value = arg_opt->get().value();
369 try {
370 return std::any_cast<T>(arg_value);
371 }
372 catch (const std::bad_any_cast& err) {
373 throw error::invalid_value_type(arg_opt->get().name(), typeid(T));
374 }

386 template <detail::c_argument_value_type T = std::string, std::convertible_to<T> U>
387 [[nodiscard]] T value_or(std::string_view arg_name, U&& default_value) const {
388 const auto arg_opt = this->_get_argument(arg_name);
389 if (not arg_opt)
390 throw error::argument_not_found(arg_name);
391
392 try {
393 const auto& arg_value = arg_opt->get().value();
394 return std::any_cast<T>(arg_value);
395 }
396 catch (const std::logic_error&) {
397 // positional: no value parsed
398 // optional: no value parsed + no predefined value
399 return T{std::forward<U>(default_value)};
400 }
401 catch (const std::bad_any_cast& err) {
402 throw error::invalid_value_type(arg_opt->get().name(), typeid(T));
403 }
404 }

413 template <detail::c_argument_value_type T = std::string>
414 [[nodiscard]] std::vector<T> values(std::string_view arg_name) const {
415 const auto arg_opt = this->_get_argument(arg_name);
416 if (not arg_opt)
417 throw error::argument_not_found(arg_name);
418
419 const auto& arg = arg_opt->get();
420
421 try {
422 if (not arg.has_parsed_values() and arg.has_value())
423 return std::vector<T>{std::any_cast<T>(arg.value())};
424
425 std::vector<T> values;
426 // TODO: use std::ranges::to after transition to C++23
427 std::ranges::copy(
428 std::views::transform(
429 arg.values(), [](const std::any& value) { return std::any_cast<T>(value); }
430 ),
431 std::back_inserter(values)
432 );
433 return values;
434 }
435 catch (const std::bad_any_cast& err) {
436 throw error::invalid_value_type(arg.name(), typeid(T));
437 }

497 void _verify_arg_name_pattern(const std::string_view arg_name) const {
498 if (arg_name.empty())
499 throw error::invalid_argument_name_pattern(
500 arg_name, "An argument name cannot be empty."
501 );
502
503 if (arg_name.front() == this->_flag_prefix_char)
504 throw error::invalid_argument_name_pattern(
505 arg_name,
506 std::format(
507 "An argument name cannot begin with a flag prefix character ({}).",
508 this->_flag_prefix_char
509 )
510 );
511
512 if (std::isdigit(arg_name.front()))
513 throw error::invalid_argument_name_pattern(
514 arg_name, "An argument name cannot begin with a digit."
515 );
516 }

523 [[nodiscard]] auto _name_match_predicate(const std::string_view arg_name) const noexcept {

532 [[nodiscard]] auto _name_match_predicate(const detail::argument_name& arg_name) const noexcept {
533 return [&arg_name](const arg_ptr_t& arg) { return arg->name().match(arg_name); };

541 [[nodiscard]] bool _is_arg_name_used(const detail::argument_name& arg_name) const noexcept {
542 const auto predicate = this->_name_match_predicate(arg_name);
543
544 if (std::ranges::find_if(this->_positional_args, predicate) != this->_positional_args.end())
545 return true;
546
547 if (std::ranges::find_if(this->_optional_args, predicate) != this->_optional_args.end())
548 return true;
549
550 return false;

559 template <detail::c_sized_range_of<std::string, detail::type_validator::convertible> AR>
560 [[nodiscard]] arg_token_list_t _tokenize(const AR& arg_range) const noexcept {
561 const auto n_args = std::ranges::size(arg_range);
562 if (n_args == 0ull)
563 return arg_token_list_t{};
564
565 arg_token_list_t toks;
566 toks.reserve(n_args);
567
568 for (const auto& arg : arg_range) {
569 std::string value = static_cast<std::string>(arg);
570 if (this->_is_flag(value)) {
571 this->_strip_flag_prefix(value);
572 toks.emplace_back(detail::argument_token::t_flag, std::move(value));
573 }
574 else {
575 toks.emplace_back(detail::argument_token::t_value, std::move(value));
576 }
577 }
578
579 return toks;
580 }
581
587 [[nodiscard]] bool _is_flag(const std::string& arg) const noexcept {
588 if (arg.starts_with(this->_flag_prefix))
589 return this->_is_arg_name_used({arg.substr(this->_primary_flag_prefxi_length)});
590
591 if (arg.starts_with(this->_flag_prefix_char))
592 return this->_is_arg_name_used({arg.substr(this->_secondary_flag_prefix_length)});
593
594 return false;
595 }
596
601 void _strip_flag_prefix(std::string& arg_flag) const noexcept {
602 if (arg_flag.starts_with(this->_flag_prefix))
603 arg_flag.erase(0, this->_primary_flag_prefxi_length);
604 else
605 arg_flag.erase(0, this->_secondary_flag_prefix_length);
606 }

613 void _parse_args_impl(const arg_token_list_t& arg_tokens) {
614 arg_token_list_iterator_t token_it = arg_tokens.begin();
615
616 this->_parse_positional_args(token_it, arg_tokens.end());
617
618 std::vector<std::string_view> dangling_values;
619 this->_parse_optional_args(token_it, arg_tokens.end(), dangling_values);
620
621 if (not dangling_values.empty())
622 throw error::argument_deduction_failure(dangling_values);
623 }
624
630 void _parse_positional_args(
631 arg_token_list_iterator_t& token_it, const arg_token_list_iterator_t& tokens_end
632 ) noexcept {
633 for (const auto& pos_arg : this->_positional_args) {
634 if (token_it == tokens_end)
635 return;
636
637 if (token_it->type == detail::argument_token::t_flag)
638 return;
639
640 pos_arg->set_value(token_it->value);
641 ++token_it;
642 }
643 }
644
654 void _parse_optional_args(
655 arg_token_list_iterator_t& token_it,
656 const arg_token_list_iterator_t& tokens_end,
657 std::vector<std::string_view>& dangling_values
658 ) {
659 std::optional<std::reference_wrapper<arg_ptr_t>> curr_opt_arg;
660
661 while (token_it != tokens_end) {
662 switch (token_it->type) {
663 case detail::argument_token::t_flag: {
664 if (not dangling_values.empty())
665 throw error::argument_deduction_failure(dangling_values);
666
667 auto opt_arg_it = std::ranges::find_if(
668 this->_optional_args, this->_name_match_predicate(token_it->value)
669 );
670
671 if (opt_arg_it == this->_optional_args.end())
672 throw error::argument_not_found(token_it->value);
673
674 curr_opt_arg = std::ref(*opt_arg_it);
675 if (not curr_opt_arg->get()->mark_used())
676 curr_opt_arg.reset();
677
678 break;
679 }
680 case detail::argument_token::t_value: {
681 if (not curr_opt_arg) {
682 dangling_values.emplace_back(token_it->value);
683 break;
684 }
685
686 if (not curr_opt_arg->get()->set_value(token_it->value))
687 curr_opt_arg.reset();
688
689 break;
690 }
691 }
692
693 ++token_it;
694 }
695 }

701 [[nodiscard]] bool _are_required_args_bypassed() const noexcept {
702 return std::ranges::any_of(this->_optional_args, [](const arg_ptr_t& arg) {
703 return arg->is_used() and arg->bypass_required_enabled();
704 });
705 }
706
711 void _verify_required_args() const {
712 for (const auto& arg : this->_positional_args)
713 if (not arg->is_used()) // ? use has_parsed_values
714 throw error::required_argument_not_parsed(arg->name());
715
716 for (const auto& arg : this->_optional_args)
717 if (arg->is_required() and not arg->has_value())
718 throw error::required_argument_not_parsed(arg->name());
719 }
720
725 void _verify_nvalues() const {
726 for (const auto& arg : this->_positional_args)
727 if (const auto nv_ord = arg->nvalues_ordering(); not std::is_eq(nv_ord))
728 throw error::invalid_nvalues(nv_ord, arg->name());
729
730 for (const auto& arg : this->_optional_args)
731 if (const auto nv_ord = arg->nvalues_ordering(); not std::is_eq(nv_ord))
732 throw error::invalid_nvalues(nv_ord, arg->name());
733 }
734
740 arg_opt_t _get_argument(std::string_view arg_name) const noexcept {
741 const auto predicate = this->_name_match_predicate(arg_name);
742
743 if (auto pos_arg_it = std::ranges::find_if(this->_positional_args, predicate);
744 pos_arg_it != this->_positional_args.end()) {
745 return std::ref(**pos_arg_it);
746 }
747
748 if (auto opt_arg_it = std::ranges::find_if(this->_optional_args, predicate);
749 opt_arg_it != this->_optional_args.end()) {
750 return std::ref(**opt_arg_it);
751 }
752
753 return std::nullopt;
754 }

761 void _print(std::ostream& os, const arg_ptr_list_t& args, const bool verbose) const noexcept {
762 if (verbose) {
763 for (const auto& arg : args)
764 os << '\n' << arg->desc(verbose).get(this->_indent_width) << '\n';
765 }
766 else {
767 std::vector<detail::argument_descriptor> descriptors;
768 descriptors.reserve(args.size());
769
770 for (const auto& arg : args)
771 descriptors.emplace_back(arg->desc(verbose));
772
773 std::size_t max_arg_name_length = 0ull;
774 for (const auto& desc : descriptors)
775 max_arg_name_length = std::max(max_arg_name_length, desc.name.length());
776
777 for (const auto& desc : descriptors)
778 os << '\n' << desc.get_basic(this->_indent_width, max_arg_name_length);
779
780 os << '\n';
781 }
782 }

805inline void add_default_argument(

826inline void add_default_argument(
827 const argument::default_optional arg_discriminator, argument_parser& arg_parser
828) noexcept {
829 switch (arg_discriminator) {
830 case argument::default_optional::help:
831 arg_parser.add_flag("help", "h")
832 .action<action_type::on_flag>(action::print_config(arg_parser, EXIT_SUCCESS))
833 .help("Display the help message");
834 break;
835
836 case argument::default_optional::input:
837 arg_parser.add_optional_argument("input", "i")
838 .required()
839 .nargs(1)
840 .action<action_type::observe>(action::check_file_exists())
841 .help("Input file path");
842 break;
843
844 case argument::default_optional::output:
845 arg_parser.add_optional_argument("output", "o").required().nargs(1).help("Output file path");
846 break;
847
848 case argument::default_optional::multi_input:
849 arg_parser.add_optional_argument("input", "i")
850 .required()
851 .nargs(ap::nargs::at_least(1))
852 .action<action_type::observe>(action::check_file_exists())
853 .help("Input files paths");
854 break;
855
856 case argument::default_optional::multi_output:
857 arg_parser.add_optional_argument("output", "o")
858 .required()
859 .nargs(ap::nargs::at_least(1))
860 .help("Output files paths");
861 break;
862 }
